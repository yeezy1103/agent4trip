\section{课题背景与目标}

\begin{frame}{研究背景}
  \begin{columns}[T]
    \begin{column}{0.55\linewidth}
      \begin{itemize}
        \item 自由行规模增长 \footnote[frame]{数据来源：中国旅游研究院，2026}
        \item 规划复杂：景点+天气+酒店+预算
        \item 规划效率低：平台割裂，信息需手动整合
        \item 大语言模型和多智能体机制为“真实数据 + 约束规划 + 结构化输出”提供了实现路径。
      \end{itemize}
    \end{column}
    \begin{column}{0.38\linewidth}
      \begin{block}{论文核心问题}
        如何面向个性化自由行需求，利用大语言模型和多智能体机制，生成一份可展示、可编辑、可导出的多日旅行计划。
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{研究目标与主要内容}
  \begin{columns}[T]
    \begin{column}{0.48\linewidth}
      \begin{block}{研究目标}
        \begin{itemize}
          \item 提升旅行计划生成效率和个性化程度。
          \item 降低模型编造景点、酒店或天气信息的风险。
          \item 将天气风险、候选 POI 和预算约束纳入规划流程。
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\linewidth}
      \begin{block}{完成内容}
        \begin{itemize}
          \item 设计多智能体旅行规划后端。
          \item 实现天气风险评估、结构化解析和安全回退。
          \item 实现前端输入、地图可视化、行程编辑和导出。
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\section{系统设计}

\begin{frame}{系统总体架构}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.6\linewidth}
      \centering
      \thesisimage{pic/3-1.pdf}{width=\linewidth,height=\textheight,keepaspectratio}
    \end{column}
    \begin{column}{0.4\linewidth}
      \begin{itemize}
        \item 前端层：需求输入、结果展示、地图可视化、编辑与导出。
        \item 后端层：FastAPI 接口、参数校验、任务调度和统一响应。
        \item Agent层：天气、景点、酒店和行程规划四类 Agent。
        \item 外部服务层：高德地图 Web 服务（后端服务）、高德地图 Web 端服务(JS API)、图片服务和大语言模型。
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{核心机制：多智能体}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.32\linewidth}
      \begin{block}{为什么不用单模型？}
        \begin{itemize}
          \item 旅行规划涉及天气、景点、酒店、路线等多类信息。
          \item 单模型一次性处理过多约束，容易遗漏条件或生成混乱结果。
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.32\linewidth}
      \begin{block}{怎么拆？}
        \begin{itemize}
          \item 天气 Agent
            \begin{itemize}
              \item 判断天气风险
            \end{itemize}
          \item 景点 Agent
            \begin{itemize}
              \item 搜索候选景点
            \end{itemize}
          \item 酒店 Agent
            \begin{itemize}
              \item 推荐住宿区域
            \end{itemize}
          \item 行程 Agent
            \begin{itemize}
              \item 生成每日安排
            \end{itemize}
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.32\linewidth}
      \begin{block}{怎么协作？}
        \begin{itemize}
          \item 按顺序执行任务。
          \item 上一阶段结果作为下一阶段上下文。
          \item 最终统一生成结构化行程。
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}

  \vspace{0.8em}
  \begin{center}
    \colorbox{softpurple}{\strut\hspace{1em}把复杂问题拆成多个可控子任务\hspace{1em}}
  \end{center}
\end{frame}

\begin{frame}{功能模块设计}
  \begin{columns}[T]
    \begin{column}{0.32\linewidth}
      \begin{block}{输入侧}
        \begin{itemize}

          \item 出行日期与天数
          \item 交通与住宿偏好
          \item 兴趣标签与额外要求
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.32\linewidth}
      \begin{block}{生成侧}
        \begin{itemize}
          \item 天气查询与风险分级
          \item 景点与酒店候选获取
          \item 结构化行程生成
          \item 预算汇总与异常回退
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.32\linewidth}
      \begin{block}{展示侧}
        \begin{itemize}
          \item 每日行程卡片
          \item 地图标记与路线展示
          \item 景点编辑与排序
          \item PNG/PDF 导出
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{关键实现一：Agent的实现\hfill{\small 为什么调用工具}}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\linewidth}
      \begin{block}{如果直接调用 API}
        \begin{itemize}
          \item 后端代码固定决定调用哪个接口。
          \item Agent 只负责润色文本，无法自主选择关键词和查询动作。
          \item 高德 API 参数多、返回结构复杂，容易把业务逻辑写散。
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\linewidth}
      \begin{block}{封装成工具后}
        \begin{itemize}
          \item Agent 根据任务和用户偏好决定是否调用、调用什么工具。
          \item 工具负责真实数据访问，降低编造景点、酒店和天气的风险。
          \item API Key、协议通信、结果格式由工具层统一封装。
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}

  \vspace{0.8em}
  \begin{center}
    \colorbox{softpurple}{\strut\hspace{1em}工具让 Agent 从“会说”变成“会查、会用真实数据规划”\hspace{1em}}
  \end{center}
\end{frame}

\begin{frame}[fragile]{关键实现一：Agent的实现\hfill{\small 工具调用底层逻辑}}
  \begin{center}
    \begin{tikzpicture}[

      \node[box, right=of parser] (tool) {ToolRegistry\\查工具};
      \node[box, below=of tool] (server) {tool.run\\执行工具};
      \node[box, left=of server] (result) {SimpleAgent\\返回工具结果};
      \node[box, left=of result] (final) {LLM\\基于工具结果生成结构化输出};

      \draw[arrow] (agent) -- (parser);
      \draw[arrow] (parser) -- (tool);
      \draw[arrow] (tool) -- (server);
      \draw[arrow] (server) -- (result);
      \draw[arrow] (result) -- (final);
    \end{tikzpicture}
  \end{center}

  \vspace{0.4em}
  \begin{block}{项目中的工具标记}
    \small
    \texttt{[TOOL\_CALL:amap\_maps\_text\_search:keywords=历史文化,city=广州]}\\
    \texttt{[TOOL\_CALL:amap\_maps\_weather:city=广州]}
  \end{block}
  \framehint{Agent 只需要输出工具调用标记；解析、进程通信、HTTP 请求和结果回传由 HelloAgents 的工具系统完成。}
\end{frame}

\begin{frame}{关键实现一：Agent的实现\hfill{\small 可调用工具与作用}}
  \small
  \begin{table}
    \centering
    \begin{tabularx}{0.96\linewidth}{@{}>{\ttfamily}lXl@{}}
      \toprule
      工具名 & 作用 & 本项目用途 \\
      \midrule
      amap\_maps\_text\_search & 按关键词和城市搜索 POI，返回名称、地址、类型、坐标等信息。 & 景点/酒店候选 \\
      amap\_maps\_weather & 查询城市天气预报，返回日期、昼夜天气、温度、风力等信息。 & 天气风险评估 \\
      maps\_geo & 地址转经纬度坐标。 & 坐标补全扩展 \\
      maps\_search\_detail & 根据 POI id 查询更详细的信息。 & 详情增强扩展 \\
      maps\_direction\_* & 步行、驾车、公交等路线规划。 & 路线优化扩展 \\
      \bottomrule
    \end{tabularx}
  \end{table}

  \framehint{当前核心流程主要使用 \texttt{amap\_maps\_text\_search} 和 \texttt{amap\_maps\_weather}；其他高德 MCP 子工具保留为后续路线规划、详情增强和坐标修正能力。}
\end{frame}

\begin{frame}{关键实现三：天气风险评估}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\linewidth}
      \begin{table}
        \small
        \centering
        \begin{tabular}{@{}lll@{}}
          \toprule
          风险因素 & 示例 & 处理策略 \\
          \midrule
          强降水/强对流 & 暴雨、雷阵雨 & 减少户外 \\
          普通降水 & 小雨、阵雨 & 优先室内 \\
          高温/低温 & 高温、低温 & 降低强度 \\
          大风/低能见度 & 大风、雾、霾 & 减少跨区 \\
          \bottomrule
        \end{tabular}
      \end{table}
    \end{column}
    \begin{column}{0.47\linewidth}
      \begin{block}{分级结果}
        \begin{itemize}
          \item \textcolor{riskgreen}{低风险}：正常安排户外活动。
          \item \textcolor{riskorange}{中风险}：安排同区域、可避雨景点。
          \item \textcolor{riskred}{高风险}：优先室内或半室内景点。
        \end{itemize}
      \end{block}
      \framehint{风险等级会同时进入景点搜索提示词、行程规划提示词和前端结果展示。}
    \end{column}
  \end{columns}
\end{frame}
